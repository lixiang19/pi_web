# 会话 LRU 池改造 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 删除顶部标签栏，重构为最多保留 5 个正式会话的 LRU 池，草稿视图独立存在且切换即丢失，每次新建都生成全新空白草稿。

**Architecture:** 引入 `useSessionLruPool` 作为工作台多会话显示的唯一真源，`WorkbenchPage` 与 `SessionTabArea` 只消费池状态，不再维护标签页。`SessionTabContent` 与 `usePerSessionChat` 解除对 tab 身份的依赖，草稿首发后通过内部 `resolvedSessionId` 切换到正式会话，再由 LRU 池接管显示与 SSE 生命周期。

**Tech Stack:** Vue 3、TypeScript、Vite、Vitest、Vue Test Utils、ESLint、vue-tsc

```ts
import { computed, ref } from "vue";

export interface LruPoolEntry {
  sessionId: string;
  lastAccessedAt: number;
  isStreaming: boolean;
}

export interface DraftViewState {
  key: string;
  cwd: string;
  parentSessionId: string;
}

const MAX_POOL_SIZE = 5;
const pool = ref<LruPoolEntry[]>([]);
const activeSessionIdState = ref<string | null>(null);
const draftViewState = ref<DraftViewState | null>(null);

const now = () => Date.now();
const createDraftKey = () => `draft:${Date.now()}-${Math.random().toString(16).slice(2)}`;

const upsertPoolEntry = (sessionId: string) => {
  const previous = pool.value.find((entry) => entry.sessionId === sessionId);
  const existing = pool.value.filter((entry) => entry.sessionId !== sessionId);
  pool.value = [
    {
      sessionId,
      lastAccessedAt: now(),
      isStreaming: previous?.isStreaming ?? false,
    },
    ...existing,
  ];
};

export const resetSessionLruPoolForTest = () => {
  pool.value = [];
  activeSessionIdState.value = null;
  draftViewState.value = null;
};

export function useSessionLruPool() {
  const activateSession = (sessionId: string) => {
    draftViewState.value = null;
    upsertPoolEntry(sessionId);
    activeSessionIdState.value = sessionId;
  };

  return {
    pool,
    draftView: computed(() => draftViewState.value),
    activeSessionId: computed(() => activeSessionIdState.value),
    isViewingDraft: computed(() => draftViewState.value !== null),
    currentViewId: computed(() =>
      draftViewState.value ? draftViewState.value.key : activeSessionIdState.value,
    ),
    activateSession,
  };
}
```

- [ ] **Step 6: 补全 LRU 实现直到第二组测试通过**

```ts
const getEvictCandidate = () =>
  [...pool.value]
    .reverse()
    .find((entry) => !entry.isStreaming)?.sessionId ?? null;

const trimPool = () => {
  while (pool.value.length > MAX_POOL_SIZE) {
    const candidate = getEvictCandidate();
    if (!candidate) {
      return;
    }
    pool.value = pool.value.filter((entry) => entry.sessionId !== candidate);
  }
};

const activateDraft = (payload?: { cwd?: string; parentSessionId?: string }) => {
  draftViewState.value = {
    key: createDraftKey(),
    cwd: payload?.cwd ?? "",
    parentSessionId: payload?.parentSessionId ?? "",
  };
  activeSessionIdState.value = null;
};

const setStreaming = (sessionId: string, streaming: boolean) => {
  pool.value = pool.value.map((entry) =>
    entry.sessionId === sessionId ? { ...entry, isStreaming: streaming } : entry,
  );
  if (!streaming) {
    trimPool();
  }
};

const activateSession = (sessionId: string) => {
  draftViewState.value = null;
  upsertPoolEntry(sessionId);
  trimPool();
  activeSessionIdState.value = sessionId;
};

const removeSession = (sessionId: string) => {
  pool.value = pool.value.filter((entry) => entry.sessionId !== sessionId);
  if (activeSessionIdState.value === sessionId) {
    activeSessionIdState.value = pool.value[0]?.sessionId ?? null;
  }
};
```

```ts
export function usePerSessionChat(sessionIdRef: Ref<string>) {
  const core = usePiChatCore();
  const resolvedSessionId = ref(sessionIdRef.value);

  watch(
    sessionIdRef,
    (nextSessionId) => {
      resolvedSessionId.value = nextSessionId;
    },
    { immediate: true },
  );

  const activeSession = computed(() =>
    core.sessions.value.find((session) => session.id === resolvedSessionId.value) ?? null,
  );

  const loadSession = async (sid: string) => {
    resolvedSessionId.value = sid;
    const snapshot = core.getCachedSessionSnapshot(sid) ?? await core.hydrateSession(sid);
    applySnapshotToSession(snapshot);
  };

  const createAndLoadSession = async (options?: {
    cwd?: string;
    title?: string;
    model?: string;
    thinkingLevel?: ThinkingLevel | null;
    parentSessionId?: string;
    agent?: string | null;
  }) => {
    const snapshot = wrapUiSessionSnapshot(await createSession(
      omitUndefined({
        cwd: options?.cwd ?? activeDraftContext.value?.cwd ?? activeSession.value?.cwd,
        title: options?.title,
        model: options?.model ?? (composer.selectedModel || undefined),
        thinkingLevel: options?.thinkingLevel ?? (composer.selectedThinkingLevel || null),
        parentSessionId: options?.parentSessionId,
        agent: options?.agent ?? (composer.selectedAgent || null),
      }),
    ));
    resolvedSessionId.value = snapshot.id;
    applySnapshotToSession(snapshot);
    return snapshot;
  };

  return {
    sessionId: computed(() => resolvedSessionId.value),
    activeSession,
    loadSession,
    createAndLoadSession,
  };
}
```

```ts
import { computed, onMounted, toRef, watch } from "vue";
import { usePerSessionChat } from "@/composables/usePerSessionChat";
import { useSessionLruPool } from "@/composables/useSessionLruPool";

const sessionIdRef = toRef(props, "sessionId");
const chat = usePerSessionChat(sessionIdRef);
const lru = useSessionLruPool();

watch(
  () => chat.sessionId.value,
  (nextSessionId, previousSessionId) => {
    if (!nextSessionId || nextSessionId === previousSessionId) {
      return;
    }
    lru.activateSession(nextSessionId);
  },
);

watch(
  [() => chat.sessionId.value, () => chat.status.value],
  ([nextSessionId, nextStatus]) => {
    if (!nextSessionId) {
      return;
    }
    lru.setStreaming(nextSessionId, nextStatus === "streaming");
  },
  { immediate: true },
);

onMounted(async () => {
  if (props.sessionId) {
    await chat.loadSession(props.sessionId);
    return;
  }

  await chat.openSessionDraft({
    cwd: props.initialCwd || undefined,
    parentSessionId: props.initialParentSessionId || undefined,
  });
});
```
